001package org.tynamo.seedentity.jpa.entities;
002
003import javax.persistence.Column;
004import javax.persistence.Entity;
005import javax.persistence.GeneratedValue;
006import javax.persistence.GenerationType;
007import javax.persistence.Id;
008
009@Entity
010public class ThingWithCustomId
011{
012        @Id
013        @GeneratedValue(strategy = GenerationType.AUTO)
014        @Column(name = "my_id")
015        private Integer id;
016
017        public Integer getId() {
018                return id;
019        }
020
021        public void setId(Integer id) {
022                this.id = id;
023        }
024}
