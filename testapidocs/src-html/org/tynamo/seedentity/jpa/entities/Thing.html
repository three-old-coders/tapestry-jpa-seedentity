012package org.tynamo.seedentity.jpa.entities;
013
014import javax.persistence.Column;
015import javax.persistence.Entity;
016import javax.persistence.GeneratedValue;
017import javax.persistence.GenerationType;
018import javax.persistence.Id;
019
020@Entity
021public class Thing {
022        @Id
023        @GeneratedValue(strategy = GenerationType.AUTO)
024        private Integer id;
025
026        @Column(unique = true)
027        private String name;
028
029        @Column(length = 300)
030        private String text;
031
032        private Integer number;
033
034        private Integer number2;
035
036        private boolean flag;
037
038        public Integer getId() {
039                return id;
040        }
041
042        /**
043         * @param id
044         *          The id to set.
045         */
046        public void setId(Integer id) {
047                this.id = id;
048        }
049
050        /**
051         * @return Returns the name.
052         */
053        public String getName() {
054                return name;
055        }
056
057        /**
058         * @param name
059         *          The name to set.
060         */
061        public void setName(String name) {
062                this.name = name;
063        }
064
065        /**
066         * @return Returns the on.
067         */
068        public boolean isFlag() {
069                return flag;
070        }
071
072        /**
073         * @param on
074         *          The on to set.
075         */
076        public void setFlag(boolean on) {
077                this.flag = on;
078        }
079
080
081        public String getText() {
082                return text;
083        }
084
085        public void setText(String text) {
086                this.text = text;
087        }
088
089        @Override
090        public boolean equals(Object o) {
091                if (this == o) return true;
092                if (o == null || getClass() != o.getClass()) return false;
093
094                Thing thing = (Thing) o;
095
096                return getId() != null ? getId().equals(thing.getId()) : thing.getId() == null;
097
098        }
099
100        @Override
101        public int hashCode() {
102                return (getId() != null ? getId().hashCode() : 0);
103        }
104
105        public String toString() {
106                return getName();
107        }
108
109        public Integer getNumber() {
110                return number;
111        }
112
113        public void setNumber(Integer number) {
114                this.number = number;
115        }
116
117        public Integer getNumber2() {
118                return number2;
119        }
120
121        public void setNumber2(Integer number2) {
122                this.number2 = number2;
123        }
124
125        public void weirdOperation() {
126                try {
127                        setNumber2(getNumber() * getNumber2());
128                } catch (RuntimeException e) {
129                        setNumber2(0);
130                }
131        }
132}
